}

\newtcolorbox{note}[1]{

before skip=20pt plus 2pt minus 2pt, % ボックスの前の空き
after skip=20pt plus 2pt minus 2pt, % ボックスの後の空き

colback=white!95, % ボックスの背景色
fonttitle=\gtfamily\bfseries, % タイトルのフォント
title=#1 % タイトルのテキスト
}
